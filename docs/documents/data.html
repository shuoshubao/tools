reserveProperties(data = {}, keys = [])
只保留对象的部分属性(删除之外的属性)

Arguments

[data] ( Object ) : 数据源
[keys] ( Array ) : 需要保留的属性列表

Returns

(*) : 修改数据源

Example

暂无
var data = { a: 1, b: 2, c: 3 }
reserveProperties(data, ['a'])
console.log(data);

// => { a: 1 };

removeProperties(data = {}, keys = [])
批量删除属性

Arguments

[data] ( Object ) : 数据源
[keys] ( Array ) : 需要删除的属性列表

Returns

(*) : 修改数据源

Example

暂无
var data = { a: 1, b: 2, c: 3 };
removeProperties(data, ['a']);
console.log(data);
// => { b: 2, c: 3 };
removeEmptyProperties(data = {})
批量删除属性值为空的属性

Arguments

[data] ( Object ) : 数据源

Returns

(*) : 修改数据源

Example

暂无
var data = { a: '', b: 0, c: false, d: null, e: { a: 0 } };
removeEmptyProperties(data, ['a']);
console.log(data);
// => { b: 0, c: false, e: { a: 0 } };
produceEmptyObject(keys = [], emptyText = '')
产生一个值全为空的对象

Arguments

[keys] ( Array ) : 属性列表
[emptyText] ( String ) : 空值

Returns

( Object ) : [值全为空的对象]

Example

暂无
produceEmptyObject(['a', 'b']);
// => { a: '', b: '' }
produceEmptyObject(['a', 'b'], null);
// => { a: null, b: null }
formatEmptyToDefault(data = {}, formater = {})
将数据中的空值替换为默认值

Arguments

data ( Object ) : 数据源
formater ( Object ) : [{ key, value }]

Returns

(*) : 修改数据源

Example

暂无
const data = {
    a: 1,
    b: null,
    c: '',
    d: ' '
};
const formater1 = {
    a: '',
    b: -1,
    c: -1
};
formatEmptyToDefault(data, formater1);
console.log(data);
// => { a: 1, b: -1, c: -1, d: ' ' }
