arrayMove(arr, fromIndex, toIndex)
数组交换位置

Arguments

(Array)
:
[数组]
(Number)
:
起始位置
(Number)
:
结束位置

Returns

(Array)
:
源数据被修改

Examples

const arr1 = [11, 22, 33, 44, 55, 66]
arrayMove(arr1, 0, 1)
console.log(arr1)
const arr2 = [11, 22, 33, 44, 55, 66]
arrayMove(arr2, 0, -1)
console.log(arr2)
reserveProperties([data], [keys])
只保留对象的部分属性(删除之外的属性)

Arguments

(Object)
:
数据源
(Array)
:
需要保留的属性列表

Returns

(*)
:
修改数据源

Examples

const data = { a: 1, b: 2, c: 3 }
reserveProperties(data, ['a'])
console.log(data);

// => { a: 1 };
removeProperties([data], [keys])
批量删除属性

Arguments

(Object)
:
数据源
(Array)
:
需要删除的属性列表

Returns

(*)
:
修改数据源

Examples

const data = { a: 1, b: 2, c: 3 };
removeProperties(data, ['a']);
console.log(data);
// => { b: 2, c: 3 };
removeEmptyProperties([data])
批量删除属性值为空的属性

Arguments

(Object)
:
数据源

Returns

(*)
:
修改数据源

Examples

const data = { a: '', b: 0, c: false, d: null, e: { a: 0 } };
removeEmptyProperties(data);
console.log(data);
// => { b: 0, c: false, e: { a: 0 } };
produceEmptyObject([keys], [emptyText])
产生一个值全为空的对象

Arguments

(Array)
:
属性列表
(String)
:
空值

Returns

(Object)
:
[值全为空的对象]

Examples

produceEmptyObject(['a', 'b']);
// => { a: '', b: '' }
produceEmptyObject(['a', 'b'], null);
// => { a: null, b: null }
formatEmptyToDefault(data, formater)
将数据中的空值替换为默认值

Arguments

(Object)
:
数据源
(Object)
:
[{ key, value }]

Returns

(*)
:
修改数据源

Examples

const data = {
    a: 1,
    b: null,
    c: '',
    d: ' '
};
const formater = {
    a: '',
    b: -1,
    c: -1
};
formatEmptyToDefault(data, formater);
console.log(data);
// => { a: 1, b: -1, c: -1, d: ' ' }
